library math {
  //Library Constants
  const PI = 3.14159;
  const E = 2.71828;

  proc square(int x) => {
    return x * x;
  }

  proc factorial(int n) => {
    if(n <= 1){
      return 1;
    }
    return n * factorial(n - 1);
  }
}
